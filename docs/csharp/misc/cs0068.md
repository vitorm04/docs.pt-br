 <span data-ttu-id="1ddff-105">Um evento em uma interface não pode ter um inicializador.</span><span class="sxs-lookup"><span data-stu-id="1ddff-105">An event in an interface cannot have an initializer.</span></span> <span data-ttu-id="1ddff-106">Para obter mais informações, consulte [interfaces](../programming-guide/interfaces/index.md).</span><span class="sxs-lookup"><span data-stu-id="1ddff-106">For more information, see [Interfaces](../programming-guide/interfaces/index.md).</span></span>

```csharp  
// CS0068.cs

interface I  
{  
   event MyDelegate d = new MyDelegate(M.f);   // CS0068  
   // try the following line instead  
   // event MyDelegate d2;  
}  
  
class M  
{  
   event MyDelegate d = new MyDelegate(M.f);  
  
   public static void f()  
   {  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
